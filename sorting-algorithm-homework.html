import random
class MyData:
    def __init__(self, data):
        self.__data = data
    
    def getData(self):
        return self.__data

    def setData(self, data):
        self.__data = data

    def randomData(self, n):
        self.__data = random.sample(range(1, n+1), n)

    def selectionSort(self):
        result = {
            "Compare": [],
            "Swap": [],
            "Count_Compare": 0,
            "Count_Swap": 0
        }
        arr = self.__data[:]
        for i in range(len(arr)):
            idx = i
            for j in range(i+1, len(arr)):
                result["Count_Compare"] += 1
                result["Compare"].append((arr[j], arr[idx]))
                if arr[j] < arr[idx]:
                    idx = j
            result["Count_Swap"] += 1
            result["Swap"].append((arr[i], arr[idx]))
            x = arr[i]
            arr[i] = arr[idx]
            arr[idx] = x
        return result

    def insertionSort(self):
        result = {
            "Compare": [],
            "Swap": [],
            "Count_Compare": 0,
            "Count_Swap": 0
        }
        arr = self.__data[:]
        for i in range(len(arr)):
            j = i
            result["Count_Compare"] += 1
            result["Compare"].append((arr[j], arr[j-1]))
            while j != 0 and (arr[j] < arr[j-1]):
                result["Count_Swap"] += 1
                result["Swap"].append((arr[j], arr[j-1]))
                arr[j], arr[j-1] = arr[j-1], arr[j]
                j -= 1
        return result 
    
    def bubbleSort(self):
        result = {
            "Compare": [],
            "Swap": [],
            "Count_Compare": 0,
            "Count_Swap": 0
        }
        dkub = self.__data[:]
        n = len(dkub)
        for i in range(n-1):
            for j in range(n-1):
                result["Count_Compare"] += 1
                result["Compare"].append((dkub[j], dkub[j+1]))
                if dkub[j+1] < dkub[j]:
                    result["Count_Swap"] += 1
                    result["Swap"].append((dkub[j], dkub[j+1]))
                    tmp = dkub[j]
                    dkub[j] = dkub[j+1]
                    dkub[j+1] = tmp
        return result
